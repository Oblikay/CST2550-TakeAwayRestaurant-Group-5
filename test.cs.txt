        [TestMethod]
        public void Order_AddItem_UpdatesItemCountAndTotalPrice()
        {
            // Arrange
            var order = new Order(1, "raj", "12345678", "Rose Hill");
            var starter = new MenuItem(10, "Spring Rolls", "Starters", 4.50m, "Crispy rolls");
            var drink = new MenuItem(20, "Cola", "Drinks", 2.00m, "Soft drink");

            // Act
            order.AddItem(starter);
            order.AddItem(drink);

            // Assert
            Assert.AreEqual(2, order.ItemCount);
            Assert.AreEqual(6.50m, order.TotalPrice);
        }

        [TestMethod]
        public void Order_DefaultStatus_IsPending()

        [TestMethod]
        public void Order_Status_CanBeUpdated()

        public void OrderBST_SearchById_ExistingOrder_ReturnsOrder()

        [TestMethod]
        public void DatabaseHelper_SaveAndLoadMenuItems_RoundTripPreservesData()
        {
            string dbPath = Path.Combine(Path.GetTempPath(), $"restaurant_test_{Guid.NewGuid()}.db");

            try
            {
                // Arrange
                var saveTree = new BinarySearchTree();
                saveTree.Insert(new MenuItem(1, "Burger", "Mains", 8.99m, "Beef burger"));
                saveTree.Insert(new MenuItem(2, "Fries", "Starters", 3.50m, "Crispy fries"));

                var db = new DatabaseHelper(dbPath);
                db.InitialiseDatabase();

                // Act
                db.SaveAllMenuItems(saveTree);

                var loadTree = new BinarySearchTree();
                db.LoadMenuItems(loadTree);

                // Assert
                Assert.AreEqual(2, loadTree.Count());

                MenuItem burger = loadTree.SearchById(1);
                MenuItem fries = loadTree.SearchById(2);

                Assert.IsNotNull(burger);
                Assert.IsNotNull(fries);
                Assert.AreEqual("Burger", burger.Name);
                Assert.AreEqual("Fries", fries.Name);
            }
            finally
            {
                if (File.Exists(dbPath))
                {
                    File.Delete(dbPath);
                }
            }
        }

        [TestMethod]
        public void DatabaseHelper_SaveAllMenuItems_UpdatesExistingItemWithSameId()
        {
            string dbPath = Path.Combine(Path.GetTempPath(), $"restaurant_test_{Guid.NewGuid()}.db");

            try
            {
                // Arrange
                var db = new DatabaseHelper(dbPath);
                db.InitialiseDatabase();

                var firstTree = new BinarySearchTree();
                firstTree.Insert(new MenuItem(5, "Pasta", "Mains", 7.50m, "Original"));
                db.SaveAllMenuItems(firstTree);

                var updatedTree = new BinarySearchTree();
                updatedTree.Insert(new MenuItem(5, "Pasta Deluxe", "Mains", 9.00m, "Updated"));

                // Act
                db.SaveAllMenuItems(updatedTree);

                var loadTree = new BinarySearchTree();
                db.LoadMenuItems(loadTree);

                // Assert
                Assert.AreEqual(1, loadTree.Count());
                MenuItem item = loadTree.SearchById(5);
                Assert.IsNotNull(item);
                Assert.AreEqual("Pasta Deluxe", item.Name);
                Assert.AreEqual(9.00m, item.Price);
            }
            finally
            {
                if (File.Exists(dbPath))
                {
                    File.Delete(dbPath);
                }
            }
        }
