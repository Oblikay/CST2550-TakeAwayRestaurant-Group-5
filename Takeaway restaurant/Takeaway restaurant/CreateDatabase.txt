--menuItems table for storing restaurant menu data--



CREATE TABLE IF NOT EXISTS MenuItems (
    Id INTEGER PRIMARY KEY,
    Name TEXT NOT NULL,
    Category TEXT NOT NULL,
    Price REAL NOT NULL,
    Description TEXT
);
