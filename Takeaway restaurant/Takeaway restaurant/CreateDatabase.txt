--menuItems table for storing restaurant menu data--



CREATE TABLE IF NOT EXISTS MenuItems (
    Id INTEGER PRIMARY KEY,
    Name TEXT NOT NULL,
    Category TEXT NOT NULL,
    Price REAL NOT NULL,
    Description TEXT
);

-- Orders Table
CREATE TABLE IF NOT EXISTS Orders (
    OrderId INTEGER PRIMARY KEY,
    CustomerName TEXT NOT NULL,
    CustomerPhone TEXT,
    CustomerAddress TEXT,
    ItemIds TEXT,
    ItemNames TEXT,
    ItemPrices TEXT,
    ItemCount INTEGER,
    TotalPrice REAL,
    Status TEXT,
    OrderDate TEXT
);
